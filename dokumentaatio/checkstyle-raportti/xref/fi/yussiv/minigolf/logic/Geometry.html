11      public static double calculateAngle(Point p1, Point p2) {
12          int vertical = p2.y - p1.y;
13          int horizontal = p2.x - p1.x;
14  
15          if (horizontal == 0) {
16              return vertical < 0 ? 180 : 0;
17          }
18          double angle = Math.toDegrees(Math.atan2(Math.abs(horizontal), Math.abs(vertical)));
19  
20          if (vertical < 0) {
21              angle += 90;
22          }
23  
24          if (horizontal > 0) {
25              angle *= -1.0;
26          }
27  
28          return angle;
29      }
