1   package fi.yussiv.minigolf.gui;
2   
3   import fi.yussiv.minigolf.domain.Ball;
4   import fi.yussiv.minigolf.domain.LevelArea;
5   import fi.yussiv.minigolf.domain.Obstacle;
6   import fi.yussiv.minigolf.domain.Wall;
7   import java.awt.BasicStroke;
8   import java.awt.Color;
9   import java.awt.Font;
10  import java.awt.Graphics;
11  import java.awt.Graphics2D;
12  import java.awt.Point;
13  import javax.swing.JPanel;
14  
15  /**
16   * The Graphical area representing the playing area.
17   */
18  public class Canvas extends JPanel {
19  
20      private static final Color GREEN = new Color(62, 142, 42);
21      private static final Color BROWN = new Color(101, 79, 14);
22      private int width;
23      private int height;
24      private Ball ball;
25      private String mouseCoords = "";
26      private LevelArea level;
27  
28      public Canvas(int width, int height, Ball ball, LevelArea level) {
29          this.width = width;
30          this.height = height;
31          this.ball = ball;
32          this.level = level;
33      }
34  
35      @Override
36      public void paintComponent(Graphics g) {
37          super.paintComponent(g);
38          g.setColor(GREEN);
39          g.fillRect(0, 0, width, height);
40  
41          int ballX = (int) ball.getX();
42          int ballY = (int) ball.getY();
43          Point target = level.getTarget();
44  
45          g.setColor(Color.BLACK);
46          g.fillOval(target.x - 9, target.y - 9, 19, 19); // point is the center
47  
48          Graphics2D g2d = (Graphics2D) g;
49  
50          for (Obstacle o : level.getObstacles()) {
51              if (o.getClass() == Wall.class) {
52                  Wall w = (Wall) o;
53                  g2d.setColor(BROWN);
54                  g2d.setStroke(new BasicStroke(w.getWidth()));
55                  g2d.drawLine(w.getStart().x, w.getStart().y, w.getEnd().x, w.getEnd().y);
56              }
57          }
58  
59          g.setColor(Color.WHITE);
60  
61          g.setFont(new Font("Arial", Font.PLAIN, 10));
62          g.drawString(mouseCoords, 480, 20);
63          g.drawString("ball:(" + ballX + "," + ballY + ")", 480, 35);
64          g.drawString(String.format("angle: %.2f", ball.getAngle()), 480, 50);
65  
66          if (ballX >= 0 && ballY >= 0) {
67              g.fillOval(ballX - 6, ballY - 6, 13, 13); // point is the center
68          } else {
69              g.setColor(Color.YELLOW);
70              g.setFont(new Font("Arial", Font.BOLD, 30));
71              g.drawString("Victory!", 280, 400);
72          }
73      }
74  
75      public void setMouseCoords(String mouseCoords) {
76          this.mouseCoords = mouseCoords;
77      }
78  
79      public void refresh() {
80          super.repaint();
81      }
82  }
