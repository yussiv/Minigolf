1   package fi.yussiv.minigolf.domain;

3   import java.awt.Point;
4   
5   /**
6    * Coordinate class using floating point values.
7    */
8   public class Position {
9   
10      private double x;
11      private double y;
12  
13      public Position(double x, double y) {
14          this.x = x;
15          this.y = y;
16      }
17  
18      public double getX() {
19          return x;
20      }
21  
22      public double getY() {
23          return y;
24      }
25  
26      @Override
27      public String toString() {
28          return "(" + x + ", " + y + ")";
29      }
30  
31      /**
32       * Method to determine if position overlaps a point within given margin.
33       * 
34       * @param point the possible overlapping point
35       * @param margin how far from the point can be to still be considered overlapping
36       */
37      public boolean overlaps(Point point, int margin) {
38          return Math.abs(x - point.x) < margin && Math.abs(y - point.y) < margin;
39      }
40  
41      @Override
42      public boolean equals(Object obj) {
43          if (!obj.getClass().equals(Position.class)) {
44              return false;
45          }
46  
47          Position other = (Position) obj;
48          return other.x == x && other.y == y;
49      }
50  
51  }
