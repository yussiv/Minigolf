1   package fi.yussiv.minigolf.domain;
2   
3   import java.awt.Point;
4   
5   /**
6    * Class representing a straight wall.
7    */
8   public class Wall implements Obstacle {
9   
10      private final Point start;
11      private final Point end;
12      private final double angle;
13      private final int width;
14      private final int length;
15  
16      public Wall(Point start, double angle, int width, int length) {
17          this.start = start;
18          this.angle = angle;
19          this.width = width;
20          this.length = length;
21          this.end = calculateEnd();
22      }
23  
24      @Override
25      public boolean overlaps(Point point, int margin) {
26  
27          // handle zero and 90 degree angles
28          if (Math.abs(angle) < 0.1) {
29              // vertical wall
30              return point.y > start.y - width / 2 - margin && point.y < end.y + width / 2 + margin && Math.abs(point.x - start.x) < width / 2 + margin;
31          } else if (Math.abs(angle - 90) < 0.1) {
32              // horizontal wall
33              return point.x > start.x - width / 2 - margin && point.x < end.x + width / 2 + margin && Math.abs(point.y - start.y) < width / 2 + margin;
34          }
35  
36          return false;
37      }
38  
39      public Point getStart() {
40          return start;
41      }
42  
43      public Point getEnd() {
44          return end;
45      }
46  
47      private Point calculateEnd() {
48          int x = (int) Math.round(start.x + Math.sin(Math.toRadians(angle)) * length);
49          int y = (int) Math.round(start.y + Math.cos(Math.toRadians(angle)) * length);
50  
51          return new Point(x, y);
52      }
53  
54      public int getWidth() {
55          return width;
56      }
57  
58      /**
59       * Wall endings have a 90 degree change in angle.
60       * 
61       * @param ball
62       * @return 
63       */
64      @Override
65      public double getAngle(Ball ball) {
66  
67          // hit start 
68          if (ball.getPosition().distance(start) < width / 2 + ball.getRadius()) {
69              if ((Math.abs(angle % 180) < 0.01 && Math.abs(ball.getAngle()) < 90)
70                      || (Math.abs(Math.abs(angle) - 90) < 0.01 && ball.getAngle() > 0)) {
71                  return this.angle + 90;
72              }
73          }
74  
75          // hit end
76          if (ball.getPosition().distance(end) < width / 2 + ball.getRadius()) {
77              if ((Math.abs(angle) - 90 < 0.01 && Math.abs(ball.getAngle()) > 90)
78                      || (Math.abs(angle) - 90 < 0.01 && ball.getAngle() < 0)) {
79                  return this.angle + 90;
80              }
81          }
82  
83          return this.angle;
84      }
85  }
