16  /**
17   * The game's main class.
18   */
19  public class Game extends Timer implements ActionListener {
20  
21      private static final double RESISTANCE_COEFFICIENT = 0.99;
22      private static final double MAX_FORCE = 100;
23      private Player player;
24      private LevelArea area;
25      private GUI gui;
26      private boolean gameOver = false;
27  
28      /**
29       * Constructor including a couple of walls.
30       */
31      public Game() {
32          super(15, null);
33          this.area = new LevelArea(600, 800);
34          this.area.addObstacle(new Wall(new Point(100, 50), 90, 10, 400));
35          this.area.addObstacle(new Wall(new Point(100, 50), 0, 10, 600));
36          this.area.addObstacle(new Wall(new Point(500, 50), 0, 10, 600));
37          this.area.addObstacle(new Wall(new Point(250, 150), 90, 20, 100));
38  
39          this.area.setStart(new Point(300, 650));
40          this.area.setTarget(new Point(300, 100));
41  
42          this.player = new Player("Player 1");
43          this.player.setBall(new Ball());
44      }
45  
46      public LevelArea getLevelArea() {
47          return area;
48      }
49  
50      /**
51       * Adds a player to the game. Also initializes a ball for said player.
52       * 
53       * @param player 
54       */
55      public void addPlayer(Player player) {
56          this.player = player;
57          Ball ball = new Ball();
58          player.setBall(ball);
59          ball.setPlayer(player);
60      }
61  
62      /**
63       * Initializes a new game. Sets ball starting location and fires up the GUI.
64       */
65      public void startGame() {
66          player.getBall().setPosition(area.getStart());
67          gui = new GUI(this);
68          SwingUtilities.invokeLater(gui);
69          addActionListener(this);
70          start();
71      }
72  
73      /**
74       * Sets the balls movement parameters for the next hit, effectively putting
75       * the ball in motion.
76       *
77       * @param force between 0 and 100
78       * @param angle -180 to 180 degrees. 0 degrees means straight up.
79       */
80      public void excecutePutt(double force, double angle) {
81          Ball ball = player.getBall();
82          if (force > MAX_FORCE) {
83              force = MAX_FORCE;
84          }
85          ball.setVelocity(force / 5);
86          ball.setAngle(angle + 180);
87      }
88  
89      /**
90       * Checks whether the ball has hit something.
91       */
92      public void maybeCollision() {
93  
94          Ball ball = player.getBall();
95  
96          // hit top or bottom
97          if (ball.getX() > area.getWidth() - 5 || ball.getX() < 5) {
98              ball.setAngle(Geometry.calculateRicochetAngle(ball.getAngle(), 0));
99          }
100 
101         // hit left or right edge
102         if (ball.getY() < 5 || ball.getY() > area.getHeight() - 5) {
103             ball.setAngle(Geometry.calculateRicochetAngle(ball.getAngle(), 90));
104         }
105 
106         // hit an obstacle
107         for (Obstacle o : area.getObstacles()) {
108             if (o.overlaps(ball.getPosition(), ball.getRadius())) {
109                 double obstacleAngle = o.getAngle(ball);
110                 double newBallAngle = Geometry.calculateRicochetAngle(ball.getAngle(), obstacleAngle);
111 
112                 ball.setAngle(newBallAngle);
113             }
114         }
115     }
116 
117     /**
118      * Check if ball is over the target location. Sets boolean indicating game is over
119      * and if so, sets a flag that the ball should not be drawn by the canvas.
120      */
121     private void maybeReachedTarget() {
122         gameOver = player.getBall().getPosition().distance(area.getTarget()) < 8;
123         if (gameOver) {
124             player.getBall().setVisible(false);
125         }
126     }
127 
128     public Player getPlayer() {
129         return player;
130     }
131 
132     /**
133      * Method to process changes in the GUI. Triggered by the timer at even
134      * intervals (15ms). Calculates the movement of the ball during the
135      * interval.
136      *
137      * @param ae
138      */
139     @Override
140     public void actionPerformed(ActionEvent ae) {
141         Ball ball = player.getBall();
142 
143         if (ball.isMoving() && !gameOver) {
144             double dx = 0, dy = 0, x, y;
145             dx = ball.getVelocity() * Math.sin(Math.toRadians(ball.getAngle()));
146             dy = ball.getVelocity() * Math.cos(Math.toRadians(ball.getAngle()));
147             x = ball.getX() + dx;
148             y = ball.getY() + dy;
149 
150             // simulate rolling resistance (=slow ball down)
151             ball.setVelocity(ball.getVelocity() * RESISTANCE_COEFFICIENT);
152 
153             ball.setPosition(x, y);
154 
155             evaluateGameState();
156         }
157         gui.refresh();
158     }
159 
160     /**
161      * Runs subroutines that check and handle changes in the game state. 
162      */
163     public void evaluateGameState() {
164         maybeCollision();
165         maybeReachedTarget();
166     }
167 
168     public boolean gameIsOver() {
169         return gameOver;
170     }
