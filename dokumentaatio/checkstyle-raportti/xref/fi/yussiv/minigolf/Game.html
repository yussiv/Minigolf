4   import fi.yussiv.minigolf.domain.Ball;
5   import fi.yussiv.minigolf.domain.LevelArea;
6   import fi.yussiv.minigolf.domain.Player;
7   import fi.yussiv.minigolf.domain.Position;

9   
10  public class Game {
11      private static final double RESISTANCE_COEFFICIENT = 0.8;
12      private Player player;
13      private LevelArea area;
14  
15      public Game(LevelArea area) {
16          this.area = area;
17      }
18  
19      LevelArea getLevelArea() {
20          return area;
21      }
22      
23      public void addPlayer(Player player) {
24          this.player = player;
25          Ball ball = new Ball();
26          player.setBall(ball);
27          ball.setPlayer(player);
28      }
29      
30      public void startGame() {
31          player.getBall().setPosition(area.getStart());
32      }
33      
34      /**
35       * 
36       * @param velocity
37       * @param angle -180 to 180, 0 == up
38       */
39      public void excecutePutt(double velocity, double angle) {
40          Ball ball = player.getBall();
41          ball.setVelocity(velocity);
42          ball.setAngle(angle);
43          int dx = 0, dy = 0;
44          while (ball.isMoving() && !targetReached()) {
45              int x = ball.getX() + (int) Math.round(ball.getVelocity() * Math.sin(angle * Math.PI / 180));
46              int y = ball.getY() + (int) Math.round(ball.getVelocity() * Math.cos(angle * Math.PI / 180));
47              
48              ball.setVelocity(ball.getVelocity() * RESISTANCE_COEFFICIENT);
49              ball.setPosition(new Position(x, y));
50          }
51      }
52  
53      public boolean targetReached() {
54          return player.getBall().getPosition().overlaps(area.getTarget(), 5);
55      }
56  }
